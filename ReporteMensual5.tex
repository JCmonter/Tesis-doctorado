\section{Actividades realizadas}

Las nociones espaciales de apilamiento aparecen por primera vez en \cite{sexton2003point} donde los autores relacionan estas propiedades con espacios cuya topología es eficiente.
En \cite{simmons2004vietoris}, se estudia la $V$-modificación, tanto de un espacio como de un marco, y verifican que la noción de apilado proporciona información sobre lo que ahí llaman
el espacio de lentes de un espacio topológico. Finalmente, en \cite{sexton2006ordinal}, se muestra que hay una interacción entre la eficiecia, el apilamiento y la $V$-modificación de algunas 
familias de subconjuntos de un espacio topológico.\\

En este reporte presentamos los resultados análogos, pero en el contexto de marcos. Los resultados originales son los siguientes:
\begin{itemize}
\item \textbf{Teorema} \cite[7.4]{sexton2006ordinal}:
Un espacio $S$ es $T_1$ y apilado si y solo si $S$ es sobrio y sus $V$-modificaciones $\mathcal{Q}S$ y $\mathcal{V}S$ son canónicamente homeomorfas.

\item \textbf{Teorema} \cite[6.2]{sexton2006ordinal}:
  Sea $S\in \Top$. Si $S$ es $T_1$ y apilado, entonces $S$ es sobrio.
\end{itemize}

Para demostrar el análogo del primer teorema necesitamos los siguientes hechos preliminares:
\begin{enumerate}
\item Consideramos a $A^\wedge$ como un espacio topológico donde los abiertos básicos de $\mathcal{O}(A^\wedge)$ están dados por 
\[
  D(x)=\{F\in A^\wedge\mid x\in F\}
\]
donde $x\in A$.
\item Para $A\in \Frm$, podemos construir un marco $VA$ como en \cite{simmons2004vietoris}. Cada elemento en $\pt(VA)$ es de la forma
\[
(F, a), \quad \text{donde } F\in A^\wedge \text{ y }a\in [v_F(0), w_F(0)]. 
\]
\item Tenemos la función continua suprayectiva $\pi\colon \pt(VA)\to A^\wedge$, dada por la asignación
\[
(F,a)\mapsto F.
\]
\item Al aplicar el funtor $\mathcal{O}(-)$ a $\pi$ obtenemos el morfismo de marcos
\[
\mathcal{O}(\pi)=\pi^{-1}\colon \mathcal{O}(A^\wedge)\to \mathcal{O}(\pt(VA)).
\]
\item Los abiertos básicos en $\pt(VA)$ están dados por 
\begin{equation}\label{fibra}
(F,a)\in [-](x)\iff x\in F\quad \text{y}\quad (F,a)\in \langle -\rangle(x)\iff x\nleq a.
\end{equation}
En particular
\begin{equation}\label{fibra2}
a\in [F](x)\iff x\in F\quad \text{y}\quad a\in \langle F\rangle(x)\iff x\nleq a.
\end{equation}
\end{enumerate}

Con todo lo anterior, podemos enunciar la siguiente proposición.

\begin{prop}
Sean $A, VA\in \Frm$ y $A^\wedge$ la familia de filtros abiertos de $A$. Las siguientes afirmaciones son ciertas:
\begin{enumerate}
\item Si $A$ es $T_1$ y apilado, entonces $\mathcal{O}(A^\wedge)\cong \mathcal{O}(\pt(VA))$ bajo la función continua $\pi$.
\item Si el morfismo $\mathcal{O}(\pi)=\pi^{-1}$ induce que $\mathcal{O}(A^\wedge)\cong \mathcal{O}(\pt(VA))$, entonces $A$ es apilado.
\item Si el morfismo $\mathcal{O}(\pi)=\pi^{-1}$ induce que $\mathcal{O}(A^\wedge)\cong \mathcal{O}(\pt(VA))$ y $A$ es espacial, entonces $A$ es $T_1$.
\end{enumerate}
\end{prop}

\begin{proof}
  $\mbox{}$
\begin{enumerate}
\item Supongamos que $A$ es $T_1$ y apilado, entonces para cada $F\in A^\wedge$ se cumple que 
\[
v_F(0)=w_F(0).
\]
Consideremos $(F,a), (G,b)\in \pt(VA)$ tales que $\pi(F,a)=F=\pi(G,a)=G$. Por $T_1$ y apilado se cumple que
\[
a=v_F(0)=v_G(0)=b.
\]
Por lo tanto $(F,a)=(G,b)$, es decir, $\pi$ es inyectiva. Al ser también suprayectiva, entonces $\pi$ es biyectiva.\\

Resta demostrar que $\pi^{-1}\colon A^\wedge\to \pt(VA)$ es continua. Para ello recordemos que por \eqref{fibra}
\[
(F,a)\in [-](x)\iff x\in F\iff (F,a)\in \pi^{-1}(D(x))\iff \pi([-](x))=D(x).
\]
Para los otros abiertos básicos, por \eqref{fibra2} tenemos que
\[
a\notin \langle F\rangle(x)\iff x\leq a\iff v_F(x)\leq a.
\]
Por $T_1$ y apilado tenemos que $a=v_F(0)$ y así
\[
v_F(0)\notin \langle F\rangle(x)\iff x\leq v_F(0)\iff v_F(x)=v_F(0).
\]
Dado que $v_F(0)\neq 1$, tenemos que 
\[
v_F(0)\notin \langle F\rangle(x)\iff x\notin F
\]
o equivalentemente 
\[
a\in \langle F\rangle(x)\iff (F, a)\in \langle -\rangle(x)\iff x\in F. 
\]
Por lo tanto, $\pi(\langle -\rangle(x))=D(x)$ y $\pi^{-1}$ es una función continua. Así, $\pi$ es un homeomorfismo y se cumple que
\[
\mathcal{O}(A^\wedge)\cong\mathcal{O}(\pt(VA)).
\]

\item Supongamos que $\mathcal{O}(A^\wedge)\cong\mathcal{O}(\pt(VA))$ bajo $\mathcal{O}(\pi)=\pi^{-1}$. De aquí que, al ser $\pi^{-1}$ biyectiva, entonces $\pi$
también lo es. Luego, si $F\in A^\wedge$, entonces $(F, v_F(0)), (F, w_F(0))\in \pt(VA)$ y así
\[
\pi(F, v_F(0))=F=\pi(F, w_F(0)).
\]
Por la inyectividad de $\pi$ tenemos que $v_F(0)=w_F(0)$ y por lo tanto $A$ es apilado.

\item Supongamos que $\mathcal{O}(A^\wedge)\cong\mathcal{O}(\pt(VA))$ y $A$ espacial. Consideremos $a\in A$ y $p\in \pt A$ tales que $p\leq a$. Por la espacialidad tenemos que 
\[
a=\bigwedge\{q\in \pt A\mid a\leq q\}
\]
para todo $a\in A$. Además, por el inciso anterior tenemos que 
\[
\mathcal{O}(A^\wedge)\cong\mathcal{O}(\pt(VA))\Rightarrow v_F(0)=w_F(0)\Rightarrow \text{ todo punto es máximo en }\pt A.
\]
Luego,
\[
p\leq a=\bigwedge\{q\in \pt A\mid a\leq q\}\leq q=p.
\]
Por lo tanto $p=a$ y $A$ es $T_1$.
\end{enumerate}
\end{proof}

Observése que en el inciso 3), basta suponer que
\[
\mathcal{V}(a)=\{q\in \pt A\mid a\leq q\}\neq \emptyset
\]
para que el resultado sea cierto. De esta manera, podría ocurrir que los resultados en los cuales pedimos que $A$ sea espacial,
basta con que $\mathcal{V}(a)\neq \emptyset$ con $a\in A$.\\

Para el análogo del segundo punto mencionado al principio de este reporte, deberíamos verificar que
\[
\text{Si }A \text{ es }T_1 \text{ y apilado }\Rightarrow A \text{ es espacial}.
\]

Considerando lo antes mencionado con respecto al conjunto $\mathcal{V}(-)$, nuestra hipótesis es que en realidad, en el caso general se cumple lo siguiente
\[
\mathbf{\text{Afirmación:}} \text{ Existen marcos }T_1\text{ y apilados que no son espaciales.}
\]

La idea para construir el ejemplo de un marco que cumpla las condiciones anteriores se sigue de lo presentado en \cite{PasekaSmarda1992}. En este artículo exhiben una familia de marcos 
Hausdorff, en el sentido de Johnstone y Shu-Hao, que no son espaciales. Para estudiar con mayor detalle los marcos apilados y fuertemente apilados, primero veremos como se comportan bajo cocientes.\\

Recordemos que para $(f\colon A\to B)\in \Frm$, $j\in NA$ y $k\in NB$, por las propiedades del funtor $N(-)$, se cumple que
\[
N(f)(j)=\bigvee\{u_{f(j(a))}\wedge v_{f(a)}\mid a\in A\}\in NA_j\quad\text{ y }\quad N(f)_*(k)=f_*\circ k\circ f\in NA,
\]
donde $f_*$ es el adjundo derecho de $f$. Siguiendo esta notación, denominamos a $N(f)(j)$ como un \emph{empuje} y a $N(f)_*(k)$ como un \emph{levantamiento}. 

\begin{prop}
Sean $A\in \Frm$ y $j\in NA$. Si $A$ es fuertemente apilado, entonces $A_j$ es fuertemente apilado.
\end{prop}

\begin{proof}
Consideremos $S=\pt A$, $S_j=\pt A_j$ y sea $F\in A_j^{\wedge}$. Así, podemos obtener los siguientes:
\[
G=j_*(F),\quad Q=S_j\setminus F,\quad R=S\setminus G,
\]
donde $j_*$ es el adjunto derecho asociado al morfismo de marcos $j^*\colon A\to A_j$ y $x\in G$ si y solo si $j(x)\in F$. Considerando las reflexiones espaciales $\mathcal{U}\colon A\to \mathcal{O}S$ y $\mathcal{U}_j\colon A_j\to \mathcal{O}S_j$ tenemos el siguiente diagrama

\[\begin{tikzcd}
	&&&& A && {\mathcal{O}S} \\
	\\
	A && {\mathcal{O}S} && {A_j} && {\mathcal{O}S_j} \\
	\\
	{A_j} && {\mathcal{O}S_j}
	\arrow["{\mathcal{U}}", from=1-5, to=1-7]
	\arrow["{j^*}"'{pos=0.7}, from=1-5, to=3-5]
	\arrow["sj", from=1-7, to=3-7]
	\arrow["{v_{G}}", from=3-1, to=1-5]
	\arrow["{\mathcal{U}}", from=3-1, to=3-3]
	\arrow["{j^*}"', from=3-1, to=5-1]
	\arrow["{[R']}"{pos=0.4}, from=3-3, to=1-7]
	\arrow["sj"{pos=0.6}, from=3-3, to=5-3]
	\arrow["{\mathcal{U}_j}"', from=3-5, to=3-7]
	\arrow["{v_F}"{pos=0.7}, from=5-1, to=3-5]
	\arrow["{\mathcal{U}_j}"', from=5-1, to=5-3]
	\arrow["{[Q']}"', from=5-3, to=3-7]
\end{tikzcd}\]
y usando los adjuntos derechos de $\mathcal{U}$ y $\mathcal{U}_j$ obtenemos
\[
v_F, \quad l=N(\mathcal{U}_j)_*([Q']),\quad m=N(j^*)(N(\mathcal{U})_*)([R']),\quad n=N(j^*)(N(\mathcal{U}_j\circ j^*)_*)([Q']) 
\]
son núcleos en $NA_j$. Veamos cuál es la relación entre todos estos núcleos.\\

Primero, por construcción, tenemos que $v_F\leq  N(\mathcal{U}_j)_*([Q'])$ y debemos verificar la otra desigualdad para que $A_j$ sea fuertemente apilado.\\

Por hipótesis, se cumple que $v_G=N(\mathcal{U})_*([R'])$. Además, para $x\in A$, se cumple que $j(v_x)\leq v_{j(x)}$ y, por definición de los núcleos ajustados asociados a $F$ y $G$, 
\begin{equation}\label{eq1}
j\circ v_G\leq v_F.
\end{equation}

Además, para $(j\colon A\to A_j)\in\Frm$ 
\[
N(j^*)(v_G)=\bigvee\{u_{j(v_G(a))}\wedge v_{j(a)}\mid a\in A\}\quad \text{y}\quad v_F=\bigvee\{u_{v_F(a)}\wedge v_a\mid a\in A_j\}. 
\]
Notemos que, por \eqref{eq1}, tenemos $u_{j(v_G)}\leq u_{v_F}$ y por lo tanto 
\begin{equation}\label{eq2}
m\leq v_F. 
\end{equation}

Para $l$ y $n$ notemos que
\[
n=\bigvee\{u_{j^*(n(x))}\wedge v_{j^*(x)}\mid x\in A\}\quad \text{y}\quad l=\bigvee\{u_{l(x)}\wedge v_x\mid x\in A_j\},
\]
donde 
\[
j^*(n(x))=j^*((j_*\circ \mathcal{U}_{j*}\circ [Q']\circ U_j\circ j^*)(x))\quad \text{y}\quad l(x)=N(\mathcal{U}_j)_*([Q'])(x).
\]

Luego
\[
\begin{split}
j^*(n(x))&=j^*(j_*(\bigvee\{a\in A\mid \mathcal{U}_j(a)\subseteq Q'\cup \mathcal{U}_j(j(x))\}))\\
&= \bigvee\{a\in A\mid \mathcal{U}_j(a)\subseteq Q'\cup \mathcal{U}_j(j(x))\}
\end{split}
\]
y ya que para $x\in A_j$ se cumple que $j(x)=x$, entonces
\[
l(x)=\bigvee\{a\in A\mid \mathcal{U}_j\subseteq Q'\cup \mathcal{U}_j(j(x))\}.
\]
Por lo tanto $j(n(x))=l(x)$.\\

Desarrollando $m(x)$ tenemos
\[
m(x)=\bigvee\{u_{j^*(\mathcal{U}_*\circ [R']\circ\mathcal{U})(x)}\wedge v_{j^*(x)}\mid x\in A\}.
\]

Dado que $A_j\subseteq A$, se cumple que $Q=R\cap A_j$ y así
\[
m(x)=\bigvee\{u_{j^*(\mathcal{U}_*\circ [Q']\circ\mathcal{U})(x)}\wedge v_x\mid x\in A_j\}.
\]

Notemos que 
\[
\begin{split}
j^*(\mathcal{U}_*\circ [Q']\circ\mathcal{U})(x)&= j^*(\bigvee\{a\in A\mid \mathcal{U}(a)\subseteq Q'\cup \mathcal{U}(x)\})\\
&=\bigvee\{j^*(a)\in A_j\mid \mathcal{U}(j^*(a))\subseteq Q'\cup \mathcal{U}(x)\}
\end{split}
\]
y como $x\in A_j$
\[
\begin{split}
\mathcal{U}(j^*(a))\subseteq Q'\cup \mathcal{U}(j^*(x))&\iff \mathcal{U}(j^*(a))\cap S_j\subseteq (Q'\cup \mathcal{U}(j^*(x)))\cap S_j\\
&\iff \mathcal{U}_j(j^*(a))\subseteq Q'\cup \mathcal{U}_j(j^*(x)),
\end{split}
\]
es decir, 
\[
\begin{split}
\bigvee\{j^*(a)\in A_j\mid \mathcal{U}(j^*(a))\subseteq Q'\cup \mathcal{U}(x)\}&=\bigvee\{j^*(a)\in A_j\mid \mathcal{U}_j(j^*(a))\subseteq Q'\cup \mathcal{U}_j(j^*(x))\}\\
&=j^*(N(\mathcal{U}_j\circ j)_*)([Q']).
\end{split}
\]
Por lo tanto 
\[
\bigvee\{u_{j^*(N(\mathcal{U}_j\circ j)_*)[Q'](x)}\wedge x\mid x\in A\}=m(x).
\]
Ya que $(N(\mathcal{U}_j\circ j)_*)([Q'])\leq j(N(\mathcal{U}_j\circ j)_*)([Q'])$ y por \eqref{eq2} 
\[
l\leq m\leq v_F\leq l
\]
y así $v_F=\mathcal{U}_{j*}\circ [Q']\circ \mathcal{U}_j$, es decir, $A_j$ es fuertemente apilado.
\end{proof}

Como todo marco fuertemente apilado es apilado, podemos afirmar lo siguiente.

\begin{cor}
Sean $A\in \Frm$ y $j\in NA$. Si $A$ es fuertemente apilado, entonces $A_j$ es apilado. 
\end{cor}

\begin{proof}
Por la proposición anterior, Si $A$ es fuertemente apilado, entonces $A_j$ es fuertemente apilado y esto implica que $A_j$ es apilado.
\end{proof}

Si $A$ es un marco Hausdorff, entonces $A$ es $T_1$. De esta manera, para poder utilizar el ejemplo de Paseka y Smarda basta que $\mathbf{(H)}$ implique apilado.

\begin{prop}
Si $A\in \Frm$ satisface $\mathbf{(H)}$, entonces $A$ es apilado.
\end{prop}

\mathcal{F}(a,b)=\{F\in \Fil(A)\mid a\in F\text{ y }b\notin F\},

\mathcal{M}\subseteq \nabla(v_{\mathcal{M}})

\pt(A_\mathcal{M})=S\setminus \mathcal{M}

\[
\pt(A_\mathcal{M})=\emptyset\quad \text{ y }\quad \pt(A_\mathcal{M})\neq \emptyset.

p\vee b=p\quad \text{o}\quad p\vee b=1  \iff b\leq p\quad \text{o}\quad b\nleq p.
\]

\text{1) } p\nleq b\;\text{ y }\; b\leq p, \qquad \text{2) } p\nleq b\; \text{ y }\;b\nleq p.

De esta manera, si se cumple que $p=v_\mathcal{M}(b)$, entonces $a\nleq p$ y $b\leq p$, es decir, $A$ es espacial. Veamos que $v_\mathcal{M}(b)=1$ nos lleva a una contradicción.\\
